server {
    listen 80;
    server_name leaderboard.ip2.space;

    # Optional: serve static files if needed
    root /root/spictank/public;
    index index.html;

    location / {
        proxy_pass http://127.0.0.1:3001;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }

    location /api/ {
        proxy_pass http://127.0.0.1:3001;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }

    # Optional: log files
    access_log /var/log/nginx/leaderboard.access.log;
    error_log /var/log/nginx/leaderboard.error.log;
}
